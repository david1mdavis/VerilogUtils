  def initialize(type, bus, name, comment)
    case(type)
    when EXP_INPUT
      @type = "input"
    when EXP_OUTPUT
      @type = "output"
    when EXP_OUTPUTREG
      @type = "output reg"
    when EXP_WIRE
      @type = "wire"
    when EXP_REG
      @type = "reg"
    when EXP_FUNCTION
      @type = "function"
    else 
      raise VarTypeException, "variable error, unknown type : #{@type}"
    end
    @bus = bus
    @name = name
    @comment = comment
  end
